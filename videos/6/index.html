import java.util.ArrayList;
import java.util.List;

public class Main {

	/**
	 * List是一种有序列表，与数组类比，但长度可变
	 */
	public static void main(String[] args) {

		List<String> list = new ArrayList<>();
		list.add("Apple");
		list.add("Pear");
		list.add("Orange");
		for (String s : list) {
			System.out.println(s);
		}

	}
}
